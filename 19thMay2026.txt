# Logical Operators

# AND
select first_name, last_name,department, salary from worker
where department = "admin" and salary > 100000;

# OR
select first_name, last_name,department, salary from worker
where department = "admin" or salary > 100000;

select first_name, last_name,department, salary from worker
where worker_id = 1 or worker_id = 7;

# IN Operator (to avoid bif lines of code in case of large querry involving or operaotr)
select worker_id, first_name, last_name, salary from worker
where worker_id  in (1,7);

# Not in operator
select worker_id, first_name, last_name, salary from worker
where worker_id  not in (1,7);

# Between Operator  (between takes range inclusive)
select worker_id, first_name, last_name, department, salary from worker
where salary between 100000 and 300000;

select worker_id, first_name, last_name, department, salary from worker
where salary between 101000 and 300000;

select worker_id, first_name, last_name, department, salary from worker
where salary not between 101000 and 300000;

select * from worker
where salary between 100000 and 500000
and department not in ('hr','admin');

# Work out - Excercise : Aggregate Functions

# 1. min() : Find the minimum salary among workers whose first name starts with 'V'.
select min(salary) from worker
where first_name like 'v%';

# 2. max() : Find the highest salary among workers in the 'Account' department.
select max(salary) from worker
where department = 'account';

# 3. count() : Find the number of workers whose salary is greater than 100000.
select count(*) from worker
where salary > 100000;

# 4. sum() : Find the total salary of workers in the 'HR' department.
select sum(salary) from worker
where department = 'hr';

# 5. avg() : Find the average salary of workers who belong to 'Admin' department and whose salary is greater than 200000.
select avg(salary) from worker
where department = 'admin' and salary > 200000;

# alias : 1) Display worker first names with the column name as EMPLOYEE_NAME.

# 2) Display the total salary of all workers with column name TOTAL_SALARY.
select sum(salary) as total_salary from worker;
